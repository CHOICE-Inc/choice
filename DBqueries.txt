create database choice_db;

--make sure your on the choice_db database before creating tables


--create tables with foreign keys

--Staff of the employees. Also if exists validated registration.
create table "staff"(
	id serial primary key,
	name varchar(50) not null,
	location varchar(20) not null,
	email varchar(50) not null
);

--login authenticated and roles of the user
create table "login"(
	id serial primary key,
	staff_id integer not null references staff(id),
	username varchar(20) not null,
	password varchar(50) not null,
	role integer not null

);

--name of clients assigned to a staff member
create table "client" (
	id serial primary key,
	staff_id int not null references staff(id),
	name varchar(50) not null
);

--locations of jobs located and desriptions
create table "job_site" (
	id serial primary key,
	business_name varchar(20) not null,
	address varchar(50) null,
	phone varchar(20) null,
	contact varchar(30) null
);

--goals assigned to a client and assigned to a case manager
create table "goal" (
	id serial primary key,
	client_id int not null references client(id),
	jobsite_id int not null references job_site(id),
	implementation_date date not null,
	review_dates varchar(50) null,
	completion_date varchar(20) not null,
	service_outcome varchar(50) not null,
	objective varchar(500) not null,
	implemented_by int not null references staff(id),
	behavior_technique varchar(200) not null,
	modifications varchar(200) not null,
	equipment varchar(50) not null,
	when_notes varchar(500) not null
);

--junction table assigning a goal with many job-sites
create table "job" (
	id serial primary key,
	jobsite_id integer not null references job_site(id),
	goal_id int not null references goal(id)
);

--ordered list of how to plan step by step
create table "steps_plan"(
	id serial primary key,
	goal_id int not null references goal(id),
	"index" int not null,
	description varchar(200) not null
);

--submitted goals
create table "goal_tracking" (
	id serial primary key,
	goal_id int not null references goal(id),
	date_tracked date not null,
	am_or_pm varchar(2) not null,
	complete_or_not varchar(20) not null,
	notes varchar(100) not null,
	additional_notes varchar(100) not null
);
